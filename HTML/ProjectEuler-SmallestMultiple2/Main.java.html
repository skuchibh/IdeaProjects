public class Main { 
 
    public static void main(String[] args) { 
        InputReader in = new InputReader(System.in); 
        OutputWriter out = new OutputWriter(System.out); 
        int k = 1, n = 19, checkk = 0; 
        while (k == 1) { 
 
            if (n % 16 == 0 && n % 12 == 0 && n % 20 == 0 && n % 14 == 0 && n % 18 == 0 && n % 11 == 0 && n % 13 == 0 && n % 15 == 0 && n % 17 == 0 && n % 19 == 0) { 
 
                out.print(n); 
                out.close(); 
                System.exit(0); 
                k = 10; 
 
 
            } 
            checkk = n + 19; 
            n = check(checkk); 
            System.out.println("n = " + n); 
 
 
        } 
        out.close(); 
 
    } 
 
    public static int check(int n) { 
        int kk = 0; 
        if (n % 16 == 0) { 
            kk = n; 
            System.out.println(kk); 
            return kk; 
        } 
        else { 
            n += 19; 
            check(n); 
        } 
        return kk; 
    } 
}
