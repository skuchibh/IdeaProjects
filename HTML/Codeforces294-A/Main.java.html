import java.io.BufferedReader; 
import java.io.IOException; 
import java.io.InputStreamReader; 
import java.io.PrintWriter; 
 
public class Main { 
    static char[][] board; 
    static int white=0, black=0, i=0; 
    static final int q = 9, r = 5, b = 3, n = 3, p = 1, k = 0; 
 
    public static void main(String[] args) throws IOException { 
 
        PrintWriter out=new PrintWriter(System.out,true); 
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in)); 
        board = new char[9][9]; 
 
        for (i = 0; i < 8; i++) { 
            board[i] = br.readLine().toCharArray(); 
        } 
        for (i = 0; i < 8; i++) { 
            for (int j = 0; j < 8; j++) { 
 
                char x = board[i][j];//if(x!='.')System.out.println(x); 
                if (x == 'q') black += q; 
                if (x == 'r') black += r; 
                if (x == 'b') black += b; 
                if (x == 'n') black += n; 
                if (x == 'p') black += p; 
                if (x == 'k') black += k; 
                if (x == 'Q') white += q; 
                if (x == 'R') white += r; 
                if (x == 'B') white += b; 
                if (x == 'N') white += n; 
                if (x == 'P') white += p; 
                if (x == 'K') white += k; 
                if (x == '.') white += 0; 
            } 
 
 
        } 
        
        if (white > black) { 
            out.print("White"); 
            out.close(); 
        } else if (white == black) { 
            out.print("Draw"); 
            out.close(); 
        } else { 
            out.print("Black"); 
            out.close(); 
        } 
    } 
}
